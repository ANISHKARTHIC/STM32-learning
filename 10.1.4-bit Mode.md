
// ================= RCC =================
#define RCC_AHB1ENR *(volatile unsigned int*)(0x40023800 + 0x30)

// ================= GPIOD =================
#define GPIOD_MODER *(volatile unsigned int*)(0x40020C00 + 0x00)
#define GPIOD_ODR   *(volatile unsigned int*)(0x40020C00 + 0x14)

// ================= PINS =================
#define RS (1<<0)   // PD0
#define EN (1<<1)   // PD1

// ================= DELAY =================
void delay(){
    for(volatile int i=0;i<30000;i++);
}

// ================= ENABLE =================
void lcd_enable(){
    GPIOD_ODR |= EN;
    delay();
    GPIOD_ODR &= ~EN;
}

// ================= SEND NIBBLE =================
void send_high_nibble(int data){
    GPIOD_ODR &= ~(0xF << 4);               // Clear PD4–PD7
    GPIOD_ODR |= ((data >> 4) << 4);        // Send HIGH nibble
    lcd_enable();
}

void send_low_nibble(int data){
    GPIOD_ODR &= ~(0xF << 4);
    GPIOD_ODR |= ((data & 0x0F) << 4);      // Send LOW nibble
    lcd_enable();
}

// ================= SEND COMMAND =================
void lcd_cmd(int cmd){
    GPIOD_ODR &= ~RS;   // Command mode

    send_high_nibble(cmd);
    send_low_nibble(cmd);

    delay();
}

// ================= SEND DATA =================
void lcd_data(int data){
    GPIOD_ODR |= RS;    // Data mode

    send_high_nibble(data);
    send_low_nibble(data);

    delay();
}

// ================= LCD INIT =================
void lcd_init(){
    delay();

    // Special sequence for 4-bit mode
    lcd_cmd(0x33);
    lcd_cmd(0x32);
    lcd_cmd(0x28); // 4-bit, 2 line
    lcd_cmd(0x0C); // Display ON
    lcd_cmd(0x01); // Clear
    lcd_cmd(0x06); // Entry mode
}

// ================= GPIO INIT =================
void gpio_init(){
    RCC_AHB1ENR |= (1<<3); // GPIOD

    // PD0–PD7 output
    GPIOD_MODER &= ~(0xFFFF);
    GPIOD_MODER |=  (0x5555);
}

// ================= MAIN =================
int main(){
    gpio_init();
    lcd_init();

    char *msg = "HELLO";

    while(*msg){
        lcd_data(*msg++);
    }

    while(1);
}
