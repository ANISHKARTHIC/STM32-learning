#define RCC_AHB1ENR *(volatile unsigned int*)(0x40023800 + 0x30)

// ================= GPIOD =================
#define GPIOD_MODER *(volatile unsigned int*)(0x40020C00 + 0x00)
#define GPIOD_ODR   *(volatile unsigned int*)(0x40020C00 + 0x14)

// ================= PIN DEFINITIONS =================
#define RS (1<<0)   // PD0
#define EN (1<<1)   // PD1

// ================= DELAY =================
void delay(){
    for(volatile int i=0;i<30000;i++);
}

// ================= ENABLE PULSE =================
void lcd_enable(){
    GPIOD_ODR |= EN;
    delay();
    GPIOD_ODR &= ~EN;
}

// ================= SEND COMMAND =================
void lcd_cmd(int cmd){
    GPIOD_ODR &= ~RS;             // RS = 0

    GPIOD_ODR &= ~(0xFF << 2);    // Clear D0–D7
    GPIOD_ODR |= (cmd << 2);      // Put command

    lcd_enable();
    delay();
}

// ================= SEND DATA =================
void lcd_data(int data){
    GPIOD_ODR |= RS;              // RS = 1

    GPIOD_ODR &= ~(0xFF << 2);
    GPIOD_ODR |= (data << 2);

    lcd_enable();
    delay();
}

// ================= LCD INIT =================
void lcd_init(){
    delay();

    lcd_cmd(0x38); // 8-bit mode
    lcd_cmd(0x0C); // Display ON
    lcd_cmd(0x01); // Clear
    lcd_cmd(0x06); // Entry mode
}

// ================= GPIO INIT =================
void gpio_init(){
    RCC_AHB1ENR |= (1<<3); // Enable GPIOD

    // Set PD0–PD9 as output
    GPIOD_MODER &= ~(0xFFFFF);
    GPIOD_MODER |=  (0x55555);
}

// ================= MAIN =================
int main(){
    gpio_init();
    lcd_init();

    char *msg = "HELLO";

    while(*msg){
        lcd_data(*msg++);
    }

    while(1);
}
