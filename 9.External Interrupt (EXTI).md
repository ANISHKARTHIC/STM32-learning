#define RCC_BASE 0x40023800UL
#define RCC_AHB1ENR *(volatile unsigned int*)(RCC_BASE + 0x30)
#define RCC_APB2ENR *(volatile unsigned int*)(RCC_BASE + 0x44)
#define GPIOA_MODER *(volatile unsigned int*)(0x40020000 + 0x00)
#define GPIOD_MODER *(volatile unsigned int*)(0x40020C00 + 0x00)
#define GPIOD_ODR   *(volatile unsigned int*)(0x40020C00 + 0x14)
#define SYSCFG_BASE 0x40013800
#define SYSCFG_EXTICR1 *(volatile unsigned int*)(SYSCFG_BASE + 0x08)
#define EXIT_BASE 0x40013C00
#define EXTI_IMR  *(volatile unsigned int*)(EXIT_BASE + 0x00)
#define EXTI_RTSR *(volatile unsigned int*)(EXIT_BASE + 0x08)
#define EXTI_PR   *(volatile unsigned int*)(EXIT_BASE + 0x14)
#define NVIC_ISER0 *(volatile unsigned int*)(0xE000E100)


void EXTI0_IRQHandler(void){
    if(EXTI_PR & (1 << 0)){
        EXTI_PR |= (1 << 0); // Clear flag

        GPIOD_ODR ^= (1 << 12); // Toggle LED
    }
}


int main(){
    RCC_AHB1ENR |= (1 << 0); // GPIOA
    RCC_AHB1ENR |= (1 << 3); // GPIOD
    // PA0 input
    GPIOA_MODER &= ~(3 << 0);
    // PD12 output
    GPIOD_MODER |= (1 << 24);
    RCC_APB2ENR |= (1 << 14); // Enable SYSCFG
    // Map PA0 → EXTI0
    SYSCFG_EXTICR1 &= ~(0xF << 0);
    // Unmask EXTI0
    EXTI_IMR |= (1 << 0);
    // Rising edge trigger
    EXTI_RTSR |= (1 << 0);
    // Enable EXTI0 interrupt in NVIC
    NVIC_ISER0 |= (1 << 6);
    while(1){
        // No polling
    }
}
