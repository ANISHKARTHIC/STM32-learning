#define RCC_BASE 0x40023800UL
#define AHB1_ENR *(volatile unsigned int*)(RCC_BASE + 0x30U)

#define GPIO_D_BASE 0x40020C00UL
#define GPIO_A_BASE 0x40020000UL
#define GPIOA_MODER *(volatile unsigned int*)(GPIO_A_BASE + 0x00U)
#define GPIOA_IDR   *(volatile unsigned int*)(GPIO_A_BASE + 0x10U)
#define GPIOD_MODER *(volatile unsigned int*)(GPIO_D_BASE + 0x00U)
#define GPIOD_ODR   *(volatile unsigned int*)(GPIO_D_BASE + 0x14U)
void led_delay(){
    for(volatile int i=0;i<300000;i++);
}

int main(void)
{
    // Enable GPIOA & GPIOD
    AHB1_ENR |= (1<<0);
    AHB1_ENR |= (1<<3);

    // PA0 as input
    GPIOA_MODER &= ~(3U << (0*2));

    // Pull-down for stability
    GPIOA_IDR &= ~(3U << (0*2));
    GPIOA_IDR |=  (2U << (0*2));

    // PD12 as output
    GPIOD_MODER &= ~(3U << (12*2));
    GPIOD_MODER |=  (1U << (12*2));

    while(1)
    {
        if(GPIOA_IDR & (1<<0))   // Button pressed
        {
            led_delay();  // debounce delay

            if(GPIOA_IDR & (1<<0))  // Confirm press
            {
                GPIOD_ODR ^= (1<<12);  // Toggle LED

                // Wait until release
                while(GPIOA_IDR & (1<<0));
            }
        }
    }
}
