#define RCC_BASE 0x40023800UL
#define AHB1_ENR *(volatile unsigned int*)(RCC_BASE + 0x30U)

#define GPIO_D_BASE 0x40020C00UL
#define GPIOD_MODER *(volatile unsigned int*)(GPIO_D_BASE + 0x00U)
#define GPIOD_ODR   *(volatile unsigned int*)(GPIO_D_BASE + 0x14U)

void led_delay(void) {
    for (int i = 0; i <= 150000; i++);
}

void mode(int n) {
    // Clear mode bits
    GPIOD_MODER &= ~(3 << (n * 2));

    // Set as output mode
    GPIOD_MODER |= (1 << (n * 2));

    // Clear output data register
    GPIOD_ODR &= ~(1 << n);
}

void led_on(int n) {
    GPIOD_ODR |= (1 << n);
    led_delay();
}

void led_off(int n) {
    GPIOD_ODR &= ~(1 << n);
    led_delay();
}

int main(void) {
    // Enable clock for GPIOD
    AHB1_ENR |= (1 << 3);

    // Set pins 12–15 as output
    mode(12);
    mode(13);
    mode(14);
    mode(15);

    while (1) {
        led_on(12);
        led_on(13);
        led_on(14);
        led_on(15);

        led_off(12);
        led_off(13);
        led_off(14);
        led_off(15);
    }
}
