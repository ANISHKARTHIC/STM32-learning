#define RCC_BASE 0x40023800UL
#define AHB1_ENR *(volatile unsigned int*)(RCC_BASE + 0x30U)

#define GPIO_D_BASE 0x40020C00UL
#define GPIOD_MODER *(volatile unsigned int*)(GPIO_D_BASE + 0x00U)
#define GPIOD_ODR   *(volatile unsigned int*)(GPIO_D_BASE + 0x14U)

void delay(int t){
    for(volatile int i=0;i<t;i++);
}

void tone(int d){
    GPIOD_ODR |= (1<<12);   // HIGH
    delay(d);

    GPIOD_ODR &= ~(1<<12);  // LOW
    delay(d);
}

int main(void)
{
    // Enable GPIOD
    AHB1_ENR |= (1<<3);

    // PD12 output
    GPIOD_MODER &= ~(3U << (12*2));
    GPIOD_MODER |=  (1U << (12*2));

    while(1)
    {
        // High pitch
        for(int i=0;i<500;i++){
            tone(200);   // small delay
        }

        // Medium pitch
        for(int i=0;i<500;i++){
            tone(600);
        }

        // Low pitch
        for(int i=0;i<500;i++){
            tone(1200);  // large delay
        }
    }
}
